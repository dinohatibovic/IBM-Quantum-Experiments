\end{itemize}

\subsection{BB84 QKD}
\begin{itemize}
    \item Alice baseline error: 0.60\%
    \item Bob QBER under eavesdropping: 49.0\%
    \item 82$\times$ degradation
\end{itemize}

\begin{itemize}
    \item Shannon entropy: 3.885 / 4.000 bits (97.1\%)
    \item $\chi^2 = 300.94$, $p \approx 0$
\end{itemize}

\subsection{QRNG}
\begin{itemize}

    \item Range: 16--254
